function updateSteve() {
  steve.frame++;
  steve.thinkTimer--;
  
  // Decide what to do
  if(steve.thinkTimer <= 0 && steve.state !== 'inspecting') {
    steve.thinkTimer = 100 + Math.random() * 200;
    
    // Pick a random node to inspect
    if(network.nodes.length > 0 && Math.random() > 0.3) {
      const node = network.nodes[Math.floor(Math.random() * network.nodes.length)];
      steve.targetX = node.x;
      steve.targetY = node.y + 50; // Stand below the node
      steve.inspectNode = node.id;
      steve.state = 'walking';
    }
  }
  
  // Movement
  const dx = steve.targetX - steve.x;
  const dy = steve.targetY - steve.y;
  const dist = Math.sqrt(dx*dx + dy*dy);
  
  if(dist > 5) {
    steve.x += (dx/dist) * steve.speed;
    steve.y += (dy/dist) * steve.speed;
    steve.facing = dx > 0 ? 1 : -1;
    steve.state = 'walking';
  } else if(steve.state === 'walking' && steve.inspectNode) {
    // Arrived at node, start inspecting
    steve.state = 'inspecting';
    steve.inspectTimer = 150;
    playSound('click');
  }
  
  // Inspecting countdown
  if(steve.state === 'inspecting') {
    steve.inspectTimer--;
    if(steve.inspectTimer <= 0) {
      steve.state = 'idle';
      steve.inspectNode = null;
      steve.thinkTimer = 50;
    }
  }
}

function generateTimeSeries(){
  network.timeSeries=[];
  for(let t=0;t<=24;t+=.25){
    const peak=Math.sin((t-6)*Math.PI/12)*.5+.5;
    const storm=t>=1&&t<=4?Math.sin((t-1)*Math.PI/3)*2.5:0;
    const step={time:t,nodes:{},links:{}};
    network.nodes.forEach(n=>{
      if(n.maxDepth>0){
        const d=Math.min(n.maxDepth*(.15+.35*peak+.6*storm),n.maxDepth*1.15);
        step.nodes[n.id]={depth:d,surcharge:d>n.maxDepth*.95};
      }
    });
    network.links.forEach(l=>{
      const f=l.diameter/20*(.3+peak+storm);
      const a=Math.PI*Math.pow(l.diameter/24/2,2);
      step.links[l.id]={flow:f,velocity:Math.min(f/a,10)};
    });
    network.timeSeries.push(step);
  }
}

// AUTO-LAYOUT
function autoLayout(){
  saveState();
  const iter=80,rep=4000,attr=.01,damp=.9;
  network.nodes.forEach(n=>{n.vx=0;n.vy=0;});
  for(let i=0;i<iter;i++){
    for(let j=0;j<network.nodes.length;j++){
      for(let k=j+1;k<network.nodes.length;k++){
        const a=network.nodes[j],b=network.nodes[k];
        const dx=b.x-a.x,dy=b.y-a.y;
        const d=Math.max(Math.sqrt(dx*dx+dy*dy),1);
        const f=rep/(d*d);
        const fx=(dx/d)*f,fy=(dy/d)*f;
        a.vx-=fx;a.vy-=fy;b.vx+=fx;b.vy+=fy;
      }
    }
    network.links.forEach(l=>{
      const a=network.nodes.find(n=>n.id===l.from);
      const b=network.nodes.find(n=>n.id===l.to);
      if(!a||!b)return;
      const dx=b.x-a.x,dy=b.y-a.y;
      const d=Math.sqrt(dx*dx+dy*dy);
      const f=(d-150)*attr;
      const fx=(dx/d)*f,fy=(dy/d)*f;
      a.vx+=fx;a.vy+=fy;b.vx-=fx;b.vy-=fy;
    });
    network.nodes.forEach(n=>{
      n.x+=n.vx;n.y+=n.vy;n.vx*=damp;n.vy*=damp;
      n.x=Math.max(50,Math.min(700,n.x));
      n.y=Math.max(50,Math.min(380,n.y));
    });
  }
  network.nodes.forEach(n=>{delete n.vx;delete n.vy;});
  playSound('layout');updateStatus('Auto-layout applied');
}

function findLongestPath(){
  const out=network.nodes.find(n=>n.type==='outfall');

  const path=[out.id];let cur=out.id;

  const nodes=state.profilePath.map(id=>network.nodes.find(n=>n.id===id)).filter(Boolean);
  const links=[];
  for(let i=0;i<state.profilePath.length-1;i++){
    const l=network.links.find(x=>(x.from===state.profilePath[i]&&x.to===state.profilePath[i+1])||(x.to===state.profilePath[i]&&x.from===state.profilePath[i+1]));
    if(l)links.push(l);
  }
  if(nodes.length<2)return;
  
  let totalLen=0;links.forEach(l=>totalLen+=l.length);
  let minE=Infinity,maxE=-Infinity;
  nodes.forEach(n=>{minE=Math.min(minE,n.invert);maxE=Math.max(maxE,n.invert+n.maxDepth);});
  const range=maxE-minE||1;
  const m={l:45,r:15,t:15,b:25};

function onMouseDown(e){
  const r=canvas.getBoundingClientRect();
  const mx=e.clientX-r.left,my=e.clientY-r.top;
  const wx=(mx-state.viewOffset.x)/state.zoom,wy=(my-state.viewOffset.y)/state.zoom;
  hideContextMenu();
  
  if(e.button===0){
    if(state.tool==='select'){
      const el=getElementAt(wx,wy);
      if(el){if(!e.shiftKey)state.selected=[];state.selected.push(el);state.isDragging=true;updateProps();}
      else{state.selected=[];updateProps();}
    }else if(state.tool==='pan'){
      state.isPanning=true;state.panStart={x:mx-state.viewOffset.x,y:my-state.viewOffset.y};
      canvas.style.cursor='grabbing';
    }else if(['junction','outfall','storage'].includes(state.tool)){
      addNode(wx,wy,state.tool);
    }else if(state.tool==='conduit'){
      handleLinkClick(wx,wy);
    }else if(state.tool==='subcatchment'){
      handleSubcatchClick(wx,wy);
    }else if(state.tool==='raingauge'){
      addRainGaugeAt(wx,wy);
    }else if(state.tool==='delete'){
      const el=getElementAt(wx,wy);if(el)deleteElement(el);
    }
  }
}

function onKeyDown(e){
  if(['INPUT','TEXTAREA','SELECT'].includes(e.target.tagName))return;
  if(e.ctrlKey||e.metaKey){
    if(e.key==='z'){e.shiftKey?redo():undo();e.preventDefault();}
    if(e.key==='y'){redo();e.preventDefault();}
    if(e.key==='c'){copySelected();e.preventDefault();}
    if(e.key==='v'){paste();e.preventDefault();}
    return;
  }
  switch(e.key.toLowerCase()){
    case'v':setTool('select');break;case'h':setTool('pan');break;
    case'j':setTool('junction');break;case'o':setTool('outfall');break;
    case's':setTool('storage');break;case'c':setTool('conduit');break;
    case'a':setTool('subcatchment');break;case'r':setTool('raingauge');break;
    case'delete':case'backspace':if(state.selected.length){state.selected.forEach(s=>deleteElement(s));state.selected=[];}break;
    case'escape':state.selected=[];state.linkStartNode=null;state.subcatchPoints=[];break;
    case'3':toggle3D();break;
    case'arrowup':state.viewOffset.y+=30;break;case'arrowdown':state.viewOffset.y-=30;break;
    case'arrowleft':state.viewOffset.x+=30;break;case'arrowright':state.viewOffset.x-=30;break;
    case'+':case'=':zoom(1.2);break;case'-':zoom(.8);break;
    case' ':togglePlay();e.preventDefault();break;
  }
}

// ELEMENT OPERATIONS
function getElementAt(x,y){
  for(const g of network.rainGauges){
    const d=Math.sqrt((x-g.x)**2+(y-g.y)**2);
    if(d<20)return{type:'raingauge',id:g.id};
  }
  for(const n of network.nodes){
    const sz=n.type==='storage'?45:n.type==='outfall'?38:32;
    if(x>n.x-sz/2&&x<n.x+sz/2&&y>n.y-sz/2&&y<n.y+sz/2)return{type:'node',id:n.id};
  }
  for(const l of network.links){
    const fr=network.nodes.find(n=>n.id===l.from),to=network.nodes.find(n=>n.id===l.to);
    if(!fr||!to)continue;
    const d=ptToLine(x,y,fr.x,fr.y,to.x,to.y);
    if(d<12)return{type:'link',id:l.id};
  }
  for(const s of network.subcatchments){
    if(s.polygon&&ptInPoly(x,y,s.polygon))return{type:'subcatchment',id:s.id};
  }
  return null;
}

function ptToLine(px,py,x1,y1,x2,y2){
  const A=px-x1,B=py-y1,C=x2-x1,D=y2-y1;
  const dot=A*C+B*D,lenSq=C*C+D*D;
  let param=lenSq?dot/lenSq:-1;
  let xx,yy;
  if(param<0){xx=x1;yy=y1;}else if(param>1){xx=x2;yy=y2;}else{xx=x1+param*C;yy=y1+param*D;}
  return Math.sqrt((px-xx)**2+(py-yy)**2);
}

function ptInPoly(x,y,poly){
  let inside=false;
  for(let i=0,j=poly.length-1;i<poly.length;j=i++){
    const xi=poly[i].x,yi=poly[i].y,xj=poly[j].x,yj=poly[j].y;
    if(((yi>y)!==(yj>y))&&(x<(xj-xi)*(y-yi)/(yj-yi)+xi))inside=!inside;
  }
  return inside;
}

  const id=genId(type==='junction'?'J':type==='outfall'?'O':'S');

    invert:100,maxDepth:type==='outfall'?0:type==='storage'?10:4,depth:0});

}

  else{if(el.id!==state.linkStartNode)addLink(state.linkStartNode,el.id);state.linkStartNode=null;}
}

function addLink(fromId,toId){
  saveState();
  const fr=network.nodes.find(n=>n.id===fromId),to=network.nodes.find(n=>n.id===toId);
  if(!fr||!to)return;
  const dx=to.x-fr.x,dy=to.y-fr.y,len=Math.round(Math.sqrt(dx*dx+dy*dy));
  const id=genId('C');
  network.links.push({id,from:fromId,to:toId,diameter:24,length:len,roughness:.013,material:'concrete',flow:0,velocity:0});
  generateTimeSeries();updateStats();playSound('place');
}

function addRainGaugeAt(x,y){
  saveState();
  const id=genId('RG');
  network.rainGauges.push({id,x,y,type:'intensity',data:[...defaultRain]});
  updateStats();playSound('place');
}

function deleteElement(el){
  saveState();
  if(el.type==='node'){
    network.links=network.links.filter(l=>l.from!==el.id&&l.to!==el.id);
    network.nodes=network.nodes.filter(n=>n.id!==el.id);
  }else if(el.type==='link'){network.links=network.links.filter(l=>l.id!==el.id);}
  else if(el.type==='subcatchment'){network.subcatchments=network.subcatchments.filter(s=>s.id!==el.id);}
  else if(el.type==='raingauge'){network.rainGauges=network.rainGauges.filter(g=>g.id!==el.id);}
  generateTimeSeries();updateStats();playSound('delete');
}

function genId(pre){
  let i=1;const ex=[...network.nodes,...network.links,...network.subcatchments,...network.rainGauges].map(e=>e.id);
  while(ex.includes(pre+i))i++;return pre+i;
}

function deleteSelected(){state.selected.forEach(s=>deleteElement(s));state.selected=[];hideContextMenu();}

// INP PARSER/GENERATOR
function parseINP(txt){
  const r={nodes:[],links:[],subcatchments:[],rainGauges:[]};
  const lines=txt.split('\n');let sec=null;const coords={};
  for(let ln of lines){
    ln=ln.trim();if(!ln||ln.startsWith(';'))continue;
    if(ln.startsWith('[')){sec=ln.replace(/[\[\]]/g,'').toUpperCase();continue;}
    const p=ln.split(/\s+/);
    if(sec==='JUNCTIONS'&&p.length>=2)r.nodes.push({id:p[0],type:'junction',x:100+r.nodes.length*80,y:150,invert:parseFloat(p[1])||0,maxDepth:parseFloat(p[2])||4,depth:0});
    if(sec==='OUTFALLS'&&p.length>=2)r.nodes.push({id:p[0],type:'outfall',x:100+r.nodes.length*80,y:150,invert:parseFloat(p[1])||0,maxDepth:0,depth:0});
    if(sec==='STORAGE'&&p.length>=2)r.nodes.push({id:p[0],type:'storage',x:100+r.nodes.length*80,y:150,invert:parseFloat(p[1])||0,maxDepth:parseFloat(p[2])||10,depth:0});
    if(sec==='CONDUITS'&&p.length>=4)r.links.push({id:p[0],from:p[1],to:p[2],length:parseFloat(p[3])||100,roughness:parseFloat(p[4])||.013,diameter:24,material:'concrete',flow:0,velocity:0});
    if(sec==='XSECTIONS'&&p.length>=3&&p[1]==='CIRCULAR'){const l=r.links.find(x=>x.id===p[0]);if(l)l.diameter=parseFloat(p[2])*12;}
    if(sec==='SUBCATCHMENTS'&&p.length>=5)r.subcatchments.push({id:p[0],rainGauge:p[1],outlet:p[2],area:parseFloat(p[3])||5,imperv:parseFloat(p[4])||50,width:parseFloat(p[5])||400,slope:parseFloat(p[6])||1,polygon:[]});
    if(sec==='RAINGAGES'&&p.length>=1)r.rainGauges.push({id:p[0],x:300,y:50,type:'intensity',data:[...defaultRain]});
    if(sec==='COORDINATES'&&p.length>=3)coords[p[0]]={x:parseFloat(p[1]),y:parseFloat(p[2])};
    if(sec==='POLYGONS'&&p.length>=3){const s=r.subcatchments.find(x=>x.id===p[0]);if(s)s.polygon.push({x:parseFloat(p[1]),y:parseFloat(p[2])});}
  }
  if(Object.keys(coords).length){
    let mnX=Infinity,mxX=-Infinity,mnY=Infinity,mxY=-Infinity;
    Object.values(coords).forEach(c=>{mnX=Math.min(mnX,c.x);mxX=Math.max(mxX,c.x);mnY=Math.min(mnY,c.y);mxY=Math.max(mxY,c.y);});
    const sc=Math.min(500/(mxX-mnX||1),280/(mxY-mnY||1),1);
    r.nodes.forEach(n=>{if(coords[n.id]){n.x=(coords[n.id].x-mnX)*sc+100;n.y=350-(coords[n.id].y-mnY)*sc;}});
    r.subcatchments.forEach(s=>{s.polygon=s.polygon.map(p=>({x:(p.x-mnX)*sc+100,y:350-(p.y-mnY)*sc}));});
  }
  return r;
}

function generateINP(){
  const t=document.getElementById('exportTitle')?.value||network.title;
  let o=`[TITLE]\n${t}\n\n[OPTIONS]\nFLOW_UNITS CFS\nFLOW_ROUTING DYNWAVE\n\n[JUNCTIONS]\n`;
  network.nodes.filter(n=>n.type==='junction').forEach(n=>{o+=`${n.id} ${n.invert} ${n.maxDepth} 0 0 0\n`;});
  o+='\n[OUTFALLS]\n';
  network.nodes.filter(n=>n.type==='outfall').forEach(n=>{o+=`${n.id} ${n.invert} FREE NO\n`;});
  if(network.nodes.filter(n=>n.type==='storage').length){o+='\n[STORAGE]\n';network.nodes.filter(n=>n.type==='storage').forEach(n=>{o+=`${n.id} ${n.invert} ${n.maxDepth} 0 FUNCTIONAL 1000 0 0\n`;});}
  o+='\n[CONDUITS]\n';
  network.links.forEach(l=>{o+=`${l.id} ${l.from} ${l.to} ${l.length} ${l.roughness} 0 0 0 0\n`;});
  o+='\n[XSECTIONS]\n';
  network.links.forEach(l=>{o+=`${l.id} CIRCULAR ${(l.diameter/12).toFixed(2)} 0 0 0 1\n`;});
  if(network.subcatchments.length){o+='\n[SUBCATCHMENTS]\n';network.subcatchments.forEach(s=>{o+=`${s.id} ${s.rainGauge||'*'} ${s.outlet} ${s.area} ${s.imperv} ${s.width} ${s.slope} 0\n`;});}
  if(network.rainGauges.length){o+='\n[RAINGAGES]\n';network.rainGauges.forEach(g=>{o+=`${g.id} INTENSITY 0:15 1.0 TIMESERIES ${g.id}_TS\n`;});}
  o+='\n[COORDINATES]\n';
  network.nodes.forEach(n=>{o+=`${n.id} ${n.x.toFixed(2)} ${n.y.toFixed(2)}\n`;});
  if(network.subcatchments.length){o+='\n[POLYGONS]\n';network.subcatchments.forEach(s=>{if(s.polygon)s.polygon.forEach(p=>{o+=`${s.id} ${p.x.toFixed(2)} ${p.y.toFixed(2)}\n`;});});}
  o+='\n[REPORT]\nNODES ALL\nLINKS ALL\n';
  return o;
}
